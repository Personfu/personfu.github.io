Sample Code Snippet

def generate_path(node, end, path, visited):
    path.append(node)
    visited.add(node)
    if node == end:
        return path
    neighbors = [n for n in range(node+1, node+4) if n <= end]
    random.shuffle(neighbors)
    for n in neighbors:
        if n not in visited:
            result = generate_path(n, end, path.copy(), visited.copy())
            if result:
                return result
    return None
