# ═══ Build CyberOS from source — like Terry Davis built TempleOS ═══

root@cyberos:~$ wget https://github.com/Personfu/CYBERWORLDSOURCECODE/archive/refs/heads/Stable.tar.gz
root@cyberos:~$ tar -xzf Stable.tar.gz && cd CYBERWORLDSOURCECODE-Stable

root@cyberos:~$ apt-get install build-essential debhelper bison check cmake flex groff \
    libbsd-dev libcurl4-openssl-dev libmaxminddb-dev libgtk-3-dev libltdl-dev \
    libluajit-5.1-dev libncurses5-dev libnet1-dev libpcap-dev libpcre2-dev libssl-dev

root@cyberos:~$ ./configure --prefix=/opt/cyberos --enable-all-modules
root@cyberos:~$ make -j$(nproc)
root@cyberos:~$ sudo make install

# Clone all CyberOS modules
root@cyberos:~$ for repo in FURY FURY0s1nt DISCORDBOT CyberWorld HackTools nmap ettercap; do
    git clone https://github.com/Personfu/$repo.git /opt/cyberos/modules/$repo
done
